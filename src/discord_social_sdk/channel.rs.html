1//! Channels, guilds, and the link between a lobby and a Discord channel.
2//!
3//! Every message on Discord is sent in a channel. [`Channel`] is the handle for
4//! one, obtained from `Client::channel`.
5//!
6//! The remaining types in this module exist for *channel linking*: a [`Lobby`]
7//! can be attached to a text channel in a Discord guild so that messages sent in
8//! one appear in the other. Building that flow means listing the user's guilds
9//! ([`GuildMinimal`]) and the channels inside them ([`GuildChannel`]), then
10//! inspecting the link from either side — [`LinkedLobby`] describes the lobby a
11//! channel points at, [`LinkedChannel`] the channel a lobby points at.
12//!
13//! [`Lobby`]: crate::lobby::Lobby
14

23handle! {
24    /// A Discord channel.
25    ///
26    /// `Message::channel_id` carries the ID of the channel a message was sent
27    /// in, and `Client::channel` turns that ID into one of these.
28    ///
29    /// # Staleness
30    ///
31    /// A handle references both the underlying data and the SDK instance.
32    /// Updates to the data are visible through existing handles without
33    /// re-creating them. If the SDK instance is destroyed while a handle is
34    /// still alive, every accessor returns a default value instead — an empty
35    /// string for string accessors, for example.
36    Channel(sys::Discord_ChannelHandle) {
37        drop: sys::Discord_ChannelHandle_Drop,
38        clone: sys::Discord_ChannelHandle_Clone,
39    }
40}
41
42impl Channel {
43    /// The ID of the channel.
44    pub fn id(&self) -> u64 {
45        // SAFETY: read-only call on a live handle.
46        unsafe { sys::Discord_ChannelHandle_Id(self.raw_ptr()) }
47    }
48
49    /// The name of the channel.
50    ///
51    /// Generally only channels in guilds have names, though Discord may generate
52    /// a display name for some others.
53    pub fn name(&self) -> String {
54        // SAFETY: the getter initialises the out-parameter and transfers
55        // ownership of the buffer to us.
56        unsafe { string::out(|out| sys::Discord_ChannelHandle_Name(self.raw_ptr(), out)) }
57    }
58
59    /// For DMs and group DMs, the user IDs of the channel's members.
60    ///
61    /// Empty for every other channel type.
62    pub fn recipients(&self) -> Vec<u64> {
63        // SAFETY: the getter initialises the span and transfers ownership of the
64        // backing array; the elements are plain integers needing no wrapping.
65        unsafe {
66            span::out(
67                |out| sys::Discord_ChannelHandle_Recipients(self.raw_ptr(), out),
68                |id| id,
69            )
70        }
71    }
72
73    /// The kind of channel this is.
74    pub fn channel_type(&self) -> ChannelType {
75        // SAFETY: read-only call on a live handle.
76        ChannelType::from_raw(unsafe { sys::Discord_ChannelHandle_Type(self.raw_ptr()) })
77    }
78}

80impl std::fmt::Debug for Channel {
81    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {

83            .field("id", &self.id())
84            .field("name", &self.name())
85            .field("channel_type", &self.channel_type())
86            .finish()
87    }

232    /// Whether the current user is able to link this channel to a lobby.
233    ///
234    /// This requires all of:
235    ///
236    /// - the channel is a guild text channel,
237    /// - the channel is not marked NSFW,
238    /// - the channel is not already linked to a different lobby,
239    /// - the user holds the Manage Channels, View Channel and Send Messages
240    ///   permissions in the channel.
241    pub fn is_linkable(&self) -> bool {
242        // SAFETY: read-only call on a live handle.
243        unsafe { sys::Discord_GuildChannel_IsLinkable(self.raw_ptr()) }
244    }

246    /// Set whether the current user is able to link this channel to a lobby.
247    pub fn set_is_linkable(&mut self, value: bool) {
248        // SAFETY: mutating call on a uniquely borrowed live handle.
249        unsafe { sys::Discord_GuildChannel_SetIsLinkable(self.as_raw_mut(), value) }
250    }

252    /// Whether the channel is "fully public": every member of the guild can view
253    /// it and send messages in it.
254    ///
255    /// Discord permits lobbies to be linked to private channels too, which
256    /// enables things like a private admin chat. But there is no permission
257    /// synchronisation between the game and Discord, so it is the game's
258    /// responsibility to restrict access to the lobby: *every* member of the
259    /// lobby can read and write the linked channel whether or not they would
260    /// have permission to do so on Discord.
261    ///
262    /// Rather than take that on, a game can use this flag to offer only fully
263    /// public channels for linking, or to show a clear warning before linking a
264    /// private one.
265    pub fn is_viewable_and_writeable_by_all_members(&self) -> bool {
266        // SAFETY: read-only call on a live handle.
267        unsafe { sys::Discord_GuildChannel_IsViewableAndWriteableByAllMembers(self.raw_ptr()) }
268    }
269
270    /// Set whether the channel is viewable and writeable by all guild members.
271    pub fn set_is_viewable_and_writeable_by_all_members(&mut self, value: bool) {
272        // SAFETY: mutating call on a uniquely borrowed live handle.
273        unsafe {
274            sys::Discord_GuildChannel_SetIsViewableAndWriteableByAllMembers(
275                self.as_raw_mut(),
276                value,
277            )
278        }
279    }
280
281    /// Information about the lobby currently linked to this channel, if any.
282    ///
283    /// Discord enforces that a channel can be linked to at most one lobby.
284    pub fn linked_lobby(&self) -> Option<LinkedLobby> {
285        let mut raw = MaybeUninit::<sys::Discord_LinkedLobby>::uninit();
286        // SAFETY: the out-parameter is only initialised when the call returns
287        // true, and ownership of the value then transfers to us.
288        unsafe {
289            if sys::Discord_GuildChannel_LinkedLobby(self.raw_ptr(), raw.as_mut_ptr()) {
290                Some(LinkedLobby::from_raw(raw.assume_init()))
291            } else {
292                None
293            }
294        }
295    }
296
297    /// Set the lobby linked to this channel.
298    ///
299    /// `None` clears the link. The argument is borrowed rather than consumed:
300    /// the SDK copies it during the call, matching the official C++ wrapper,
301    /// which keeps ownership of its own value across the same call.
302    pub fn set_linked_lobby(&mut self, value: Option<&LinkedLobby>) {
303        let ptr = match value {
304            Some(v) => v.raw_ptr(),
305            None => std::ptr::null_mut(),
306        };
307        // SAFETY: `ptr` is either null (meaning "clear") or points to a live
308        // handle that outlives the call, which only reads through it.
309        unsafe { sys::Discord_GuildChannel_SetLinkedLobby(self.as_raw_mut(), ptr) }
310    }
