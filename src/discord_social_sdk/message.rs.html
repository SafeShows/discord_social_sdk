1//! Chat messages, their non-text attachments, and DM conversation summaries.
2//!
3//! The SDK supports two kinds of chat: one-on-one conversations between two
4//! users, and chat inside a lobby. Both arrive as [`Message`] handles, and
5//! [`Message::channel`] tells you which context a given message came from.
6//!
7//! # Syncing with Discord
8//!
9//! Messages sent through the SDK sometimes also appear in the Discord client:
10//! for one-on-one chat when at least one participant is a full Discord user, and
11//! for lobby chat when the lobby is linked to a Discord channel. The sender must
12//! also not be banned on the Discord side.
13//!
14//! # Legal disclosures
15//!
16//! The first time a user sends an in-game message that will also show up in
17//! Discord, the SDK injects a synthetic message explaining that to the user.
18//! Those are identified by [`Message::disclosure_type`] returning `Some`.
19//! Discord encourages games to restyle, reword or translate these rather than
20//! rendering them verbatim — or to skip rendering them entirely.
21//!
22//! # History
23//!
24//! The SDK keeps the 25 most recent messages per channel in memory and has no
25//! access to anything sent before it connected. An existing [`Message`] keeps
26//! working after the SDK has discarded the message for being too old; you just
27//! cannot obtain a new handle for it.
28//!
29//! # Unrenderable content
30//!
31//! Messages can carry images, videos, embeds, polls and so on that a game is not
32//! expected to render. [`Message::additional_content`] describes that content so
33//! the game can show a small notice pointing the user to Discord.
34//!
35//! See the [message resource documentation] for more on Discord's message model.
36//!
37//! [message resource documentation]: https://discord.com/developers/docs/resources/message
38

49/// Call a getter of the shape `void G(self, Discord_Properties* out)` and adopt
50/// the result as a map.
51///
52/// The SDK allocates the key and value arrays as one block, and the individual
53/// strings belong to that same allocation — so they are copied out rather than
54/// freed individually, and the block is released with `Discord_FreeProperties`.
55///
56/// # Safety
57///
58/// `f` must fully initialise the out-parameter and transfer ownership of it.
59unsafe fn properties_out<F>(f: F) -> HashMap<String, String>
60where
61    F: FnOnce(*mut sys::Discord_Properties),
62{
63    let mut raw = MaybeUninit::<sys::Discord_Properties>::uninit();
64    f(raw.as_mut_ptr());
65    // SAFETY: `f` is contracted to have initialised the out-parameter.
66    let props = unsafe { raw.assume_init() };
67
68    let mut map = HashMap::new();
69    if !props.keys.is_null() && !props.values.is_null() {
70        for i in 0..props.size {
71            // SAFETY: both arrays hold `props.size` initialised `Discord_String`s,
72            // and `view` copies out of them without taking ownership.
73            let (key, value) = unsafe {
74                (
75                    string::view(*props.keys.add(i)),
76                    string::view(*props.values.add(i)),
77                )
78            };
79            map.insert(key, value);
80        }
81    }
82    // SAFETY: `props` was just handed to us by the SDK and is released exactly once.
83    unsafe { sys::Discord_FreeProperties(props) };
84    map
85}

87handle! {
88    /// Non-text content attached to a message.
89    ///
90    /// Describes content that likely cannot be rendered in game — images,
91    /// videos, embeds, polls and similar — so the game can show a notice that
92    /// there is more to see on Discord.
93    AdditionalContent(sys::Discord_AdditionalContent) {
94        init: sys::Discord_AdditionalContent_Init,
95        drop: sys::Discord_AdditionalContent_Drop,
96        clone: sys::Discord_AdditionalContent_Clone,
97    }
98}
99
100impl AdditionalContent {
101    /// Render an [`AdditionalContentType`] as a human-readable string.
102    pub fn type_to_string(content_type: AdditionalContentType) -> String {
103        // SAFETY: the getter fully initialises the out-parameter and transfers
104        // ownership of the buffer to us.
105        unsafe {
106            string::out(|out| {
107                sys::Discord_AdditionalContent_TypeToString(content_type.into_raw(), out)
108            })
109        }
110    }
111
112    /// The kind of additional content in the message.
113    ///
114    /// Named `content_type` because `type` is a Rust keyword.
115    pub fn content_type(&self) -> AdditionalContentType {
116        // SAFETY: read-only call on an initialised handle.
117        let raw = unsafe { sys::Discord_AdditionalContent_Type(self.raw_ptr()) };
118        AdditionalContentType::from_raw(raw)
119    }
120
121    /// Set the kind of additional content.
122    pub fn set_content_type(&mut self, value: AdditionalContentType) {
123        // SAFETY: setter on an initialised handle we own.
124        unsafe { sys::Discord_AdditionalContent_SetType(self.as_raw_mut(), value.into_raw()) }
125    }
126
127    /// The name of the poll or thread, when the additional content is one of those.
128    pub fn title(&self) -> Option<String> {
129        // SAFETY: read-only call; the out-parameter is only initialised — and
130        // then owned by us — when the call returns `true`.
131        unsafe { string::out_opt(|out| sys::Discord_AdditionalContent_Title(self.raw_ptr(), out)) }
132    }
133
134    /// Set the title, or clear it with `None`.
135    pub fn set_title(&mut self, value: Option<&str>) {
136        let ptr = self.as_raw_mut();
137        string::with_opt(value, |s| {
138            // SAFETY: the SDK copies the string during the call, so borrowing is sound.
139            unsafe { sys::Discord_AdditionalContent_SetTitle(ptr, s) }
140        })
141    }
142
143    /// How many pieces of additional content the message carries.
144    ///
145    /// Useful for rendering a notice such as "2 additional images".
146    pub fn count(&self) -> u8 {
147        // SAFETY: read-only call on an initialised handle.
148        unsafe { sys::Discord_AdditionalContent_Count(self.raw_ptr()) }
149    }
150
151    /// Set the number of pieces of additional content.
152    pub fn set_count(&mut self, value: u8) {
153        // SAFETY: setter on an initialised handle we own.
154        unsafe { sys::Discord_AdditionalContent_SetCount(self.as_raw_mut(), value) }
155    }
156}

158impl PartialEq for AdditionalContent {
159    /// Compares each field of the additional content for equality.
160    fn eq(&self, other: &Self) -> bool {
161        // SAFETY: both handles are initialised; `Equals` only reads them.
162        unsafe { sys::Discord_AdditionalContent_Equals(self.raw_ptr(), other.as_raw()) }
163    }
164}

176handle! {
177    /// A single message received by the SDK.
178    ///
179    /// Handles hold a reference to both the underlying data and the SDK
180    /// instance, so changes to the message are visible through an existing
181    /// handle without re-creating it. If the SDK instance is destroyed while a
182    /// handle is still alive, every accessor returns a default value (an empty
183    /// string, zero, `None`) rather than failing.
184    ///
185    /// Note: while the SDK lets you send messages on a user's behalf, you must
186    /// only do so in response to a user action — never automatically.
187    ///
188    /// Messages are only ever handed back by the SDK, so there is no constructor.
189    Message(sys::Discord_MessageHandle) {
190        drop: sys::Discord_MessageHandle_Drop,
191        clone: sys::Discord_MessageHandle_Clone,
192    }
193}
