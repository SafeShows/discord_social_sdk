1//! OAuth2 authorization: arguments, PKCE code challenge/verifier pairs.
2//!
3//! Everything the SDK does on a user's behalf requires an OAuth2 access token.
4//! This module holds the parameter objects for the flows that obtain one; the
5//! functions that actually run those flows live on `Client`.
6//!
7//! # The flow the SDK models
8//!
9//! 1. **Authorize.** Build [`AuthorizationArgs`] — at minimum a scope string,
10//!    which for most games is one of the client's default scope constants — and
11//!    pass it to `Client::authorize`. The SDK opens Discord (or a browser),
12//!    runs a loopback web server to catch the redirect, and hands back an
13//!    *authorization code*. That code is not a token.
14//! 2. **Exchange.** The code must be traded for an access token. Public clients
15//!    can do this in-process with `Client::get_token`, which requires the PKCE
16//!    pair described below. Apps with a backend should instead perform the
17//!    exchange server-side, which is also what lets provisional accounts later
18//!    "upgrade" into full Discord accounts.
19//! 3. **Refresh.** Access tokens obtained this way expire (7 days for the
20//!    device flow); the accompanying refresh token does not currently expire
21//!    and is used to mint new access tokens.
22//!
23//! # PKCE
24//!
25//! When the token exchange happens inside the game process there is no client
26//! secret to prove the exchange came from whoever started the flow. PKCE closes
27//! that gap: `Client::create_authorization_code_verifier` generates an
28//! [`AuthorizationCodeVerifier`], whose
29//! [`challenge`](AuthorizationCodeVerifier::challenge) goes into
30//! [`AuthorizationArgs::set_code_challenge`] and whose
31//! [`verifier`](AuthorizationCodeVerifier::verifier) goes into the later token
32//! exchange. This crate never generates these values itself — the SDK does.
33//!
34//! # Device flow
35//!
36//! On consoles, smart TVs and other limited-input devices,
37//! [`DeviceAuthorizationArgs`] feeds `Client::get_token_from_device`, which
38//! shows a short code the user enters on a second device instead of having them
39//! type credentials in place.
40//!
41//! # Secrets
42//!
43//! Values here that must never reach logs, telemetry or crash reports:
44//!
45//! - the **code verifier** ([`AuthorizationCodeVerifier::verifier`]) — anyone
46//!   holding it alongside an intercepted authorization code can redeem a token;
47//! - the **state** and **nonce** ([`AuthorizationArgs::state`],
48//!   [`AuthorizationArgs::nonce`]) — anti-CSRF and anti-replay values whose
49//!   whole security property is being unpredictable.
50//!
51//! The [`Debug`] impls in this module redact all three. The *challenge* is a
52//! hash of the verifier, travels in the clear as part of the authorization URL,
53//! and is safe to print.
54

62handle! {
63    /// The challenge half of a PKCE code verification pair.
64    ///
65    /// A hash of the secret verifier, sent to Discord when the authorization
66    /// flow begins. It is not secret and may be logged.
67    ///
68    /// Normally obtained from [`AuthorizationCodeVerifier::challenge`] rather
69    /// than built by hand; the SDK generates matching pairs.
70    AuthorizationCodeChallenge(sys::Discord_AuthorizationCodeChallenge) {
71        init: sys::Discord_AuthorizationCodeChallenge_Init,
72        drop: sys::Discord_AuthorizationCodeChallenge_Drop,
73        clone: sys::Discord_AuthorizationCodeChallenge_Clone,
74    }
75}

77impl AuthorizationCodeChallenge {
78    /// The method used to generate the challenge.
79    ///
80    /// The only method the SDK uses is SHA-256.
81    pub fn method(&self) -> CodeChallengeMethod {
82        // SAFETY: `raw_ptr` yields a valid initialised handle; this getter only reads.
83        CodeChallengeMethod::from_raw(unsafe {
84            sys::Discord_AuthorizationCodeChallenge_Method(self.raw_ptr())
85        })
86    }
87
88    /// Set the method used to generate the challenge.
89    pub fn set_method(&mut self, value: CodeChallengeMethod) {
90        // SAFETY: `as_raw_mut` yields a valid initialised handle; the enum is a plain value.
91        unsafe {
92            sys::Discord_AuthorizationCodeChallenge_SetMethod(self.as_raw_mut(), value.into_raw())
93        }
94    }
95
96    /// The challenge value.
97    pub fn challenge(&self) -> String {
98        // SAFETY: the getter fully initialises the out-parameter and transfers ownership of it.
99        unsafe {
100            string::out(|out| sys::Discord_AuthorizationCodeChallenge_Challenge(self.raw_ptr(), out))
101        }
102    }
103
104    /// Set the challenge value.
105    pub fn set_challenge(&mut self, value: &str) {
106        // SAFETY: the SDK copies the string during the call, so borrowing `value` is sound.
107        unsafe {
108            sys::Discord_AuthorizationCodeChallenge_SetChallenge(
109                self.as_raw_mut(),
110                string::borrow(value),
111            )
112        }
113    }
114}

125handle! {
126    /// A PKCE code verification pair: the secret verifier and its challenge.
127    ///
128    /// Produced by `Client::create_authorization_code_verifier`. There is
129    /// deliberately no constructor — the SDK generates the pair, and this crate
130    /// implements no cryptography of its own.
131    ///
132    /// # Secrets
133    ///
134    /// [`verifier`](Self::verifier) is a secret. Do not log it, do not persist
135    /// it beyond the lifetime of the flow, and do not send it anywhere other
136    /// than the token exchange. The [`Debug`] impl redacts it.
137    AuthorizationCodeVerifier(sys::Discord_AuthorizationCodeVerifier) {
138        drop: sys::Discord_AuthorizationCodeVerifier_Drop,
139        clone: sys::Discord_AuthorizationCodeVerifier_Clone,
140    }
141}

143impl AuthorizationCodeVerifier {
144    /// The challenge half of the pair.
145    ///
146    /// Pass this to [`AuthorizationArgs::set_code_challenge`] when starting the
147    /// authorization flow.
148    pub fn challenge(&self) -> AuthorizationCodeChallenge {
149        let mut raw = MaybeUninit::<sys::Discord_AuthorizationCodeChallenge>::uninit();
150        // SAFETY: the getter fully initialises the out-parameter and transfers ownership of the
151        // handle to us, so wrapping it in an owning `AuthorizationCodeChallenge` is correct.
152        unsafe {
153            sys::Discord_AuthorizationCodeVerifier_Challenge(self.raw_ptr(), raw.as_mut_ptr());
154            AuthorizationCodeChallenge::from_raw(raw.assume_init())
155        }
156    }
157
158    /// Set the challenge half of the pair.
159    ///
160    /// The SDK copies the challenge, so `value` stays owned by the caller.
161    pub fn set_challenge(&mut self, value: &AuthorizationCodeChallenge) {
162        // SAFETY: both handles are valid and initialised. The SDK clones the challenge rather
163        // than adopting the pointer — the official C++ wrapper passes a temporary here and lets
164        // it destruct immediately after the call.
165        unsafe {
166            sys::Discord_AuthorizationCodeVerifier_SetChallenge(self.as_raw_mut(), value.raw_ptr())
167        }
168    }
169
170    /// The verifier half of the pair.
171    ///
172    /// **This value is a secret.** It is presented during the token exchange to
173    /// prove the exchange comes from whoever began the authorization flow. Never
174    /// log it or share it.
175    pub fn verifier(&self) -> String {
176        // SAFETY: the getter fully initialises the out-parameter and transfers ownership of it.
177        unsafe {
178            string::out(|out| sys::Discord_AuthorizationCodeVerifier_Verifier(self.raw_ptr(), out))
179        }
180    }
181
182    /// Set the verifier half of the pair.
183    ///
184    /// Only useful for restoring a pair the SDK generated earlier. Do not invent
185    /// verifier values.
186    pub fn set_verifier(&mut self, value: &str) {
187        // SAFETY: the SDK copies the string during the call, so borrowing `value` is sound.
188        unsafe {
189            sys::Discord_AuthorizationCodeVerifier_SetVerifier(
190                self.as_raw_mut(),
191                string::borrow(value),
192            )
193        }
194    }
195}

197impl std::fmt::Debug for AuthorizationCodeVerifier {
198    /// Redacts the verifier; the challenge is printed in full.
199    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
200        f.debug_struct("AuthorizationCodeVerifier")
201            .field("challenge", &self.challenge())
202            .field("verifier", &"<redacted>")
203            .finish()
204    }

439impl std::fmt::Debug for AuthorizationArgs {
440    /// Redacts `state` and `nonce`, reporting only whether each is present.
441    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
442        fn redacted(value: &Option<String>) -> &'static str {
443            match value {
444                Some(_) => "<redacted>",
445                None => "None",
446            }
447        }
448
449        f.debug_struct("AuthorizationArgs")
450            .field("client_id", &self.client_id())
451            .field("scopes", &self.scopes())
452            .field("state", &redacted(&self.state()))
453            .field("nonce", &redacted(&self.nonce()))
454            .field("code_challenge", &self.code_challenge())
455            .field("integration_type", &self.integration_type())
456            .field("custom_scheme_param", &self.custom_scheme_param())
457            .finish()
458    }
459}
