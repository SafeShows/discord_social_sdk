1//! Converting `Discord_*Span` out-parameters into owned Rust [`Vec`]s.
2//!
3//! A span is a `{ ptr, size }` pair. When the SDK returns one, ownership of
4//! *both* the array and every element in it transfers to the caller: each
5//! element must be adopted individually, then the backing array released with
6//! `Discord_Free`. Elements are moved out rather than cloned, so no extra
7//! allocation happens per item.
8
9use discord_social_sdk_sys as sys;
10use std::mem::MaybeUninit;
11
12/// Call a span-returning getter and adopt every element.
13///
14/// `wrap` takes each raw element by value and must produce a wrapper that owns
15/// it — typically `|raw| unsafe { Type::from_raw(raw) }`.
16///
17/// # Safety
18///
19/// `f` must fully initialise the out-parameter and transfer ownership of the
20/// array and its elements.
21pub(crate) unsafe fn out<S, E, T, F, W>(f: F, wrap: W) -> Vec<T>
22where
23    S: SpanLayout<Elem = E>,
24    F: FnOnce(*mut S),
25    W: Fn(E) -> T,
26{
27    let mut raw = MaybeUninit::<S>::uninit();
28    f(raw.as_mut_ptr());
29    let span = unsafe { raw.assume_init() };
30
31    let (ptr, size) = span.parts();
32    if ptr.is_null() || size == 0 {
33        if !ptr.is_null() {
34            unsafe { sys::Discord_Free(ptr.cast()) };
35        }
36        return Vec::new();
37    }
38
39    let mut out = Vec::with_capacity(size);
40    for i in 0..size {
41        // Move the element out; the SDK will not touch it again.
42        let elem = unsafe { std::ptr::read(ptr.add(i)) };
43        out.push(wrap(elem));
44    }
45    // Free the array itself, not the elements — those are ours now.
46    unsafe { sys::Discord_Free(ptr.cast()) };
47    out
48}

50/// Structural access to a `Discord_*Span`.
51///
52/// Implemented via [`impl_span`] for each concrete span type, since bindgen
53/// generates them as unrelated structs with identical shape.
54pub(crate) trait SpanLayout {
55    type Elem;
56    fn parts(&self) -> (*mut Self::Elem, usize);
57}
58
59macro_rules! impl_span {
60    ($($span:ty => $elem:ty),* $(,)?) => {
61        $(
62            impl $crate::span::SpanLayout for $span {
63                type Elem = $elem;
64                fn parts(&self) -> (*mut $elem, usize) {
65                    (self.ptr, self.size)
66                }
67            }
68        )*
69    };
70}

87/// Adopt a span delivered by value as a callback argument.
88///
89/// Callback spans are **owned**, exactly like the out-param case: the official
90/// C++ wrapper adopts every element as `DiscordObjectState::Owned` and then
91/// frees the backing array. Treating one as borrowed leaks every element plus
92/// the array, so this moves the elements out and frees the array, same as
93/// [`out`] — it differs only in how the span arrives.
94///
95/// # Safety
96///
97/// `span` must be a span the SDK just transferred to us, and must not be used
98/// again afterwards.
99pub(crate) unsafe fn take<S, E, T, W>(span: S, wrap: W) -> Vec<T>
100where
101    S: SpanLayout<Elem = E>,
102    W: Fn(E) -> T,
103{
104    let (ptr, size) = span.parts();
105    if ptr.is_null() || size == 0 {
106        if !ptr.is_null() {
107            unsafe { sys::Discord_Free(ptr.cast()) };
108        }
109        return Vec::new();
110    }
111
112    let mut out = Vec::with_capacity(size);
113    for i in 0..size {
114        // Move the element out; the SDK will not touch it again.
115        let elem = unsafe { std::ptr::read(ptr.add(i)) };
116        out.push(wrap(elem));
117    }
118    // Free the array itself, not the elements — those are ours now.
119    unsafe { sys::Discord_Free(ptr.cast()) };
120    out
121}
