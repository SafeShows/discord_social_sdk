1//! Sending, editing, and observing chat messages.
2//!
3//! The SDK carries two kinds of chat, both of which arrive as [`Message`]
4//! handles:
5//!
6//! - **User messages** — one-on-one direct messages between the current user and
7//!   one recipient, sent with [`Client::send_user_message`]. A DM can only be
8//!   sent when the two users are both online in the game and have not blocked
9//!   each other, or are friends, or share a Discord server and have DM'd there
10//!   before.
11//! - **Lobby messages** — messages sent to every member of a lobby with
12//!   [`Client::send_lobby_message`]. If the lobby is linked to a Discord
13//!   channel, the message is relayed to that channel too.
14//!
15//! Message content is capped at 2,000 characters and may contain Discord's
16//! [message formatting] markup.
17//!
18//! # History
19//!
20//! [`Client::message`] reads the SDK's in-memory cache — the 25 most recent
21//! messages per channel. To load more, [`Client::fetch_lobby_messages`] and
22//! [`Client::fetch_user_messages`] go out to Discord's API and return up to 200
23//! messages from the last 72 hours, newest first.
24//!
25//! # Notifications
26//!
27//! When the Discord desktop app is running on the same machine, the user hears
28//! its notification for a message they can already see in game. Call
29//! [`Client::set_showing_chat`] whenever in-game chat is shown or hidden so
30//! Discord can suppress the duplicate.
31//!
32//! [message formatting]: https://discord.com/developers/docs/reference#message-formatting
33

42/// Build a `Discord_Properties` borrowing Rust memory and run `f` with it.
43///
44/// `Discord_Properties` is two parallel arrays of `Discord_String`. The SDK
45/// copies both the arrays and the strings they point at while the call runs —
46/// the official C++ wrapper allocates a temporary copy and frees it the instant
47/// the call returns — so borrowing the caller's `&str`s is sound as long as the
48/// backing [`Vec`]s outlive the call. Keeping them in scope here is what
49/// guarantees that.
50///
51/// The result must never be passed to `Discord_FreeProperties`: nothing in it
52/// was allocated by the SDK.
53fn with_properties<T, F>(metadata: &[(&str, &str)], f: F) -> T
54where
55    F: FnOnce(sys::Discord_Properties) -> T,
56{
57    let mut keys: Vec<sys::Discord_String> =
58        metadata.iter().map(|(k, _)| string::borrow(k)).collect();
59    let mut values: Vec<sys::Discord_String> =
60        metadata.iter().map(|(_, v)| string::borrow(v)).collect();
61    let properties = sys::Discord_Properties {
62        size: keys.len(),
63        keys: keys.as_mut_ptr(),
64        values: values.as_mut_ptr(),
65    };
66    // `keys` and `values` are still owned by this frame, so the pointers stay
67    // valid for the whole of `f`.
68    f(properties)
69}

71/// Trampoline for `Discord_Client_SendUserMessageCallback`.
72///
73/// Shared by every send site; on success it reports the new message's id.
74unsafe extern "C" fn send_tramp<F>(
75    result: *mut sys::Discord_ClientResult,
76    message_id: u64,
77    userdata: *mut c_void,
78) where
79    F: FnOnce(Result<u64>) + 'static,
80{
81    // SAFETY: `userdata` is the boxed `F` installed alongside this trampoline.
82    unsafe { callback::dispatch_once::<F>(userdata, |f| f(to_result(result).map(|()| message_id))) }
83}

85/// Trampoline for callbacks that report only success or failure.
86unsafe extern "C" fn result_tramp<F>(result: *mut sys::Discord_ClientResult, userdata: *mut c_void)
87where
88    F: FnOnce(Result<()>) + 'static,
89{
90    // SAFETY: `userdata` is the boxed `F` installed alongside this trampoline.
91    unsafe { callback::dispatch_once::<F>(userdata, |f| f(to_result(result))) }
92}

94/// Trampoline for the two message-fetching callbacks.
95///
96/// The `Discord_MessageHandleSpan` is **owned**: the SDK transfers the array and
97/// every handle in it, exactly as the official C++ wrapper does when it adopts
98/// each element as `DiscordObjectState::Owned` and then frees the array. Each
99/// element is therefore moved out rather than cloned, and the array released.
100/// Treating the span as borrowed would leak every message in it.
101unsafe extern "C" fn messages_tramp<F>(
102    result: *mut sys::Discord_ClientResult,
103    messages: sys::Discord_MessageHandleSpan,
104    userdata: *mut c_void,
105) where
106    F: FnOnce(Result<Vec<Message>>) + 'static,
107{
108    // SAFETY: the span is transferred to us — the C++ wrapper adopts every element
109    // as owned and then frees the array — so elements are moved out, not cloned.
110    unsafe {
111        callback::dispatch_once::<F>(userdata, |f| {
112            // Claimed before the result is inspected: the SDK transfers the span
113            // and every handle in it regardless of outcome, so taking it only on
114            // success would leak the array and all its messages on every failure.
115            let messages = span::take(messages, |raw| Message::from_raw(raw));
116            f(to_result(result).map(|()| messages))
117        })
118    }
119}

124    /// Send a direct message to `recipient_id`.
125    ///
126    /// The content is capped at 2,000 characters and may contain Discord's
127    /// message formatting markup. On success the callback receives the new
128    /// message's id.
129    ///
130    /// A DM can only be delivered when one of the following holds:
131    ///
132    /// - both users are online in the game and have not blocked each other,
133    /// - the users are friends, or
134    /// - the users share a Discord server and have DM'd each other on Discord.
135    ///
136    /// Only send on behalf of a user in response to a user action.
137    ///
138    /// Completes through [`run_callbacks`](crate::run_callbacks).
139    pub fn send_user_message<F>(&mut self, recipient_id: u64, content: &str, callback: F)
140    where
141        F: FnOnce(Result<u64>) + 'static,
142    {
143        // SAFETY: the content string is copied by the SDK during the call.
144        unsafe {
145            sys::Discord_Client_SendUserMessage(
146                self.as_raw_mut(),
147                recipient_id,
148                string::borrow(content),
149                Some(send_tramp::<F>),
150                callback::free_fn::<Option<F>>(),
151                callback::once_userdata(callback),
152            )
153        }
154    }

295    /// Look up a cached message by id.
296    ///
297    /// The SDK keeps the 25 most recent messages per channel in memory. Anything
298    /// older, or sent before the SDK started, yields `None`.
299    pub fn message(&self, message_id: u64) -> Option<Message> {
300        let mut raw = MaybeUninit::<sys::Discord_MessageHandle>::uninit();
301        // SAFETY: read-only call; the handle is initialised and transferred to
302        // us only when the call returns `true`.
303        unsafe {
304            if sys::Discord_Client_GetMessageHandle(self.raw_ptr(), message_id, raw.as_mut_ptr()) {
305                Some(Message::from_raw(raw.assume_init()))
306            } else {
307                None
308            }
309        }
310    }

369    /// Fetch a summary of every DM conversation the current user has.
370    ///
371    /// Each [`UserMessageSummary`] names the other user and the id of the most
372    /// recent message in that conversation; the messages themselves are loaded
373    /// with [`fetch_user_messages`](Self::fetch_user_messages).
374    pub fn fetch_user_message_summaries<F>(&mut self, callback: F)
375    where
376        F: FnOnce(Result<Vec<UserMessageSummary>>) + 'static,
377    {
378        unsafe extern "C" fn tramp<F>(
379            result: *mut sys::Discord_ClientResult,
380            summaries: sys::Discord_UserMessageSummarySpan,
381            userdata: *mut c_void,
382        ) where
383            F: FnOnce(Result<Vec<UserMessageSummary>>) + 'static,
384        {
385            // SAFETY: the span is owned — the C++ wrapper adopts each element and
386            // frees the array — so summaries are moved out rather than cloned.
387            unsafe {
388                callback::dispatch_once::<F>(userdata, |f| {
389                    // Claimed before the result is inspected: the SDK transfers this
390                    // payload regardless of outcome, so taking it only on success
391                    // would leak it on every failed request.
392                    let summaries = span::take(summaries, |raw| UserMessageSummary::from_raw(raw));
393                    f(to_result(result).map(|()| summaries))
394                })
395            }
396        }
397        // SAFETY: the SDK owns the boxed closure and frees it via `free_fn`.
398        unsafe {
399            sys::Discord_Client_GetUserMessageSummaries(
400                self.as_raw_mut(),
401                Some(tramp::<F>),
402                callback::free_fn::<Option<F>>(),
403                callback::once_userdata(callback),
404            )
405        }
406    }

420    /// Open a message in the Discord client.
421    ///
422    /// Useful when a message carries content the game cannot render — wire this
423    /// to the click handler of whatever prompt offers to show it in Discord.
424    ///
425    /// `on_merge_required` fires when the user is on a provisional account that
426    /// must be merged into a real Discord account before the message can be
427    /// opened; drive the merge flow from there. It is separate from `callback`,
428    /// which reports whether the open itself succeeded.
429    pub fn open_message_in_discord<M, F>(
430        &mut self,
431        message_id: u64,
432        on_merge_required: M,
433        callback: F,
434    ) where
435        M: FnMut() + 'static,
436        F: FnOnce(Result<()>) + 'static,
437    {
438        unsafe extern "C" fn merge_tramp<M: FnMut() + 'static>(userdata: *mut c_void) {
439            // SAFETY: `userdata` is the boxed `M` installed below.
440            unsafe { callback::dispatch_mut::<M>(userdata, |f| f()) }
441        }
442        // SAFETY: the SDK owns both boxed closures and frees each through the
443        // `free_fn` paired with it.
444        unsafe {
445            sys::Discord_Client_OpenMessageInDiscord(
446                self.as_raw_mut(),
447                message_id,
448                Some(merge_tramp::<M>),
449                callback::free_fn::<M>(),
450                callback::persistent_userdata(on_merge_required),
451                Some(result_tramp::<F>),
452                callback::free_fn::<Option<F>>(),
453                callback::once_userdata(callback),
454            )
455        }
456    }
