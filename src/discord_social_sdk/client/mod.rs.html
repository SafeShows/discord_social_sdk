1//! The [`Client`] — the entry point to the SDK.
2//!
3//! A `Client` owns the connection to Discord and is the receiver for nearly
4//! every operation the SDK offers. Its surface is large, so it is split across
5//! several files that each add an `impl Client` block:
6//!
7//! - this module — construction, connection lifecycle, status, logging, version
8//! - `auth` — OAuth2, tokens, provisional accounts
9//! - `voice` — calls, audio devices, capture and playback settings
10//! - `lobby` — creating, joining and observing lobbies
11//! - `messaging` — sending and reading messages
12//! - `relationships` — friends, blocking, friend requests
13//! - `presence` — rich presence and activity invites
14//!
15//! # Lifecycle
16//!
17//! A client must be given an application id, connected, and then driven by
18//! [`run_callbacks`](crate::run_callbacks). It is not usable until its status
19//! reaches [`ClientStatus::Ready`].
20//!
21//! ```no_run
22//! use discord_social_sdk::{Client, enums::ClientStatus, run_callbacks};
23//!
24//! let mut client = Client::new();
25//! client.set_application_id(1234567890);
26//! client.on_status_changed(|status, error, detail| {
27//!     if let Some(error) = error {
28//!         eprintln!("connection error: {error:?} ({detail})");
29//!     }
30//!     if status == ClientStatus::Ready {
31//!         println!("ready");
32//!     }
33//! });
34//! client.connect();
35//!
36//! loop {
37//!     run_callbacks();
38//!     std::thread::sleep(std::time::Duration::from_millis(16));
39//! }
40//! ```

189    ///
190    /// See the [module documentation](self) for the lifecycle and for where each
191    /// group of methods is defined.

302    /// Observe connection status changes.
303    ///
304    /// `error` is `None` while the transition is not a failure. `error_detail`
305    /// carries a gateway close code when the SDK has one; see Discord's
306    /// [gateway close event codes].
307    ///
308    /// Replaces any previously registered handler.
309    ///
310    /// [gateway close event codes]: https://discord.com/developers/docs/topics/opcodes-and-status-codes#gateway-gateway-close-event-codes
311    pub fn on_status_changed<F>(&mut self, callback: F)
312    where
313        F: FnMut(ClientStatus, Option<ClientError>, i32) + 'static,
314    {
315        unsafe extern "C" fn tramp<F>(
316            status: sys::Discord_Client_Status,
317            error: sys::Discord_Client_Error,
318            detail: i32,
319            userdata: *mut std::ffi::c_void,
320        ) where
321            F: FnMut(ClientStatus, Option<ClientError>, i32) + 'static,
322        {
323            // SAFETY: `userdata` is the boxed `F` installed below and still owned by the SDK.
324            unsafe {
325                callback::dispatch_mut::<F>(userdata, |f| {
326                    let error = match ClientError::from_raw(error) {
327                        ClientError::None => None,
328                        other => Some(other),
329                    };
330                    f(ClientStatus::from_raw(status), error, detail)
331                })
332            }
333        }
334        // SAFETY: the SDK stores the boxed closure and releases it via `free_fn::<F>()`.
335        unsafe {
336            sys::Discord_Client_SetStatusChangedCallback(
337                self.as_raw_mut(),
338                Some(tramp::<F>),
339                callback::free_fn::<F>(),
340                callback::persistent_userdata(callback),
341            )
342        }
343    }

345    /// Receive the SDK's log messages at or above `min_severity`.
346    ///
347    /// Unlike most handlers this one *adds* to the set of log sinks rather than
348    /// replacing the previous one.
349    ///
350    /// # When the closure is released is not guaranteed
351    ///
352    /// Log sinks are the one callback whose teardown the SDK does not promise,
353    /// and the two SDK builds measurably disagree:
354    ///
355    /// - the **release** library frees the closure once the [`Client`] is dropped
356    ///   and the queue is pumped, like any other callback;
357    /// - the **debug** library never frees it — not on drop, not on
358    ///   [`reset_callbacks`](crate::reset_callbacks) — so it and everything it
359    ///   captures live until the process exits.
360    ///
361    /// Write handlers that are correct either way: keep captures small, and do
362    /// not capture credentials or anything whose lifetime you need to control.
363    pub fn add_log_callback<F>(&mut self, min_severity: LoggingSeverity, callback: F)
364    where
365        F: FnMut(&str, LoggingSeverity) + 'static,
366    {
367        // SAFETY: the SDK owns the boxed closure and frees it via `free_fn`; the
368        // message string is transferred to us and is freed by `take`.
369        unsafe {
370            sys::Discord_Client_AddLogCallback(
371                self.as_raw_mut(),
372                Some(log_tramp::<F>),
373                callback::free_fn::<F>(),
374                callback::persistent_userdata(callback),
375                min_severity.into_raw(),
376            )
377        }
378    }

505/// Shared trampoline for the two log-callback registrations, which have
506/// identical signatures.
507unsafe extern "C" fn log_tramp<F>(
508    message: sys::Discord_String,
509    severity: sys::Discord_LoggingSeverity,
510    userdata: *mut std::ffi::c_void,
511) where
512    F: FnMut(&str, LoggingSeverity) + 'static,
513{
514    // SAFETY: `userdata` is the boxed `F`; `message` is transferred to us and freed by `take`.
515    unsafe {
516        callback::dispatch_mut::<F>(userdata, |f| {
517            let text = string::take(message);
518            f(&text, LoggingSeverity::from_raw(severity))
519        })
520    }
521}
