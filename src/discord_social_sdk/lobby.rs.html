1//! Lobbies and lobby membership.
2//!
3//! A lobby is an arbitrary, developer-controlled group of users that can
4//! communicate with each other. Lobbies are created and managed either through
5//! Discord's [server APIs] or, without any server-side component, through
6//! `Client::create_or_join_lobby`.
7//!
8//! [server APIs]: https://docs.discord.com/developers/resources/lobby
9//!
10//! # Metadata
11//!
12//! Both [`Lobby`] and [`LobbyMember`] carry developer-supplied metadata: a flat
13//! set of string key/value pairs. The SDK returns it as a `Discord_Properties`
14//! parallel array whose buffers we must release with `Discord_FreeProperties`;
15//! this module copies every key and value into an owned [`HashMap`] and frees
16//! the SDK's memory before returning.
17

29/// Copy a `Discord_Properties` value into an owned map and release the SDK's
30/// buffers.
31///
32/// The SDK allocates the key and value arrays *and* every string inside them.
33/// `Discord_FreeProperties` releases all of it in one go, so the strings are
34/// read with a borrowing copy rather than by taking ownership, which would
35/// double free.
36///
37/// # Safety
38///
39/// `raw` must be a fully initialised `Discord_Properties` whose ownership the
40/// SDK just transferred to us. Calling this twice on the same value double frees.
41unsafe fn take_properties(raw: sys::Discord_Properties) -> HashMap<String, String> {
42    let mut out = HashMap::with_capacity(raw.size);
43    if !raw.keys.is_null() && !raw.values.is_null() {
44        for i in 0..raw.size {
45            // SAFETY: the SDK guarantees both arrays hold `raw.size` initialised
46            // strings, each valid until `Discord_FreeProperties` runs below.
47            let (key, value) = unsafe {
48                (
49                    string::view(*raw.keys.add(i)),
50                    string::view(*raw.values.add(i)),
51                )
52            };
53            out.insert(key, value);
54        }
55    }
56    // SAFETY: `raw` came from the SDK and has not been freed yet.
57    unsafe { sys::Discord_FreeProperties(raw) };
58    out
59}
60
61/// Call an SDK getter of the shape `void Get(self, Discord_Properties* out)`.
62///
63/// # Safety
64///
65/// `f` must fully initialise the out-parameter and transfer ownership of it.
66unsafe fn properties_out<F>(f: F) -> HashMap<String, String>
67where
68    F: FnOnce(*mut sys::Discord_Properties),
69{
70    let mut raw = MaybeUninit::<sys::Discord_Properties>::uninit();
71    f(raw.as_mut_ptr());
72    // SAFETY: `f` initialised the out-parameter per this function's contract.
73    unsafe { take_properties(raw.assume_init()) }
74}

76handle! {
77    /// A single member of a [`Lobby`].
78    ///
79    /// Members are not removed automatically when they close the game or
80    /// temporarily disconnect, so a member may exist while
81    /// [`connected`](LobbyMember::connected) is `false`.
82    LobbyMember(sys::Discord_LobbyMemberHandle) {
83        drop: sys::Discord_LobbyMemberHandle_Drop,
84        clone: sys::Discord_LobbyMemberHandle_Clone,
85    }
86}

88impl LobbyMember {
89    /// Whether this user is allowed to link a channel to the lobby.
90    ///
91    /// Under the hood this checks the `CanLinkLobby` member flag, which can only
92    /// be set through the server API's `add_lobby_member`. It exists so a game
93    /// can restrict channel linking to, say, the clan or guild leader.
94    pub fn can_link_lobby(&self) -> bool {
95        // SAFETY: read-only call on a live handle.
96        unsafe { sys::Discord_LobbyMemberHandle_CanLinkLobby(self.raw_ptr()) }
97    }
98
99    /// Whether the user is currently connected to the lobby.
100    pub fn connected(&self) -> bool {
101        // SAFETY: read-only call on a live handle.
102        unsafe { sys::Discord_LobbyMemberHandle_Connected(self.raw_ptr()) }
103    }
104
105    /// The user ID of the lobby member.
106    pub fn id(&self) -> u64 {
107        // SAFETY: read-only call on a live handle.
108        unsafe { sys::Discord_LobbyMemberHandle_Id(self.raw_ptr()) }
109    }
110
111    /// Developer-supplied metadata for this member.
112    ///
113    /// A common use is to store the game's internal user ID alongside the
114    /// Discord one, so every member of a lobby knows the mapping for each user.
115    pub fn metadata(&self) -> HashMap<String, String> {
116        // SAFETY: the getter fully initialises the out-parameter and transfers
117        // ownership of the properties to us.
118        unsafe { properties_out(|out| sys::Discord_LobbyMemberHandle_Metadata(self.raw_ptr(), out)) }
119    }
120
121    /// The full user record for this member, if the SDK has one.
122    pub fn user(&self) -> Option<User> {
123        let mut raw = MaybeUninit::<sys::Discord_UserHandle>::uninit();
124        // SAFETY: the out-parameter is only initialised when the call returns
125        // true, and ownership of the handle then transfers to us.
126        unsafe {
127            if sys::Discord_LobbyMemberHandle_User(self.raw_ptr(), raw.as_mut_ptr()) {
128                Some(User::from_raw(raw.assume_init()))
129            } else {
130                None
131            }
132        }
133    }
134}

146handle! {
147    /// A lobby: a developer-controlled group of users that can chat and talk
148    /// together.
149    ///
150    /// # Lifetime
151    ///
152    /// Lobbies are ephemeral. One that has been idle — no user actively
153    /// connected — for its auto-delete timeout (5 minutes by default, up to 7
154    /// days when configured through the server API) is removed. A lobby linked
155    /// to a Discord channel is never auto-deleted.
156    ///
157    /// # Limits
158    ///
159    /// A lobby may hold at most 1,000 members, and each user may be in at most
160    /// 200 lobbies per game. Voice calls should stay far below that ceiling —
161    /// Discord recommends around 25 participants or fewer.
162    ///
163    /// # Channel linking
164    ///
165    /// A lobby can be linked to a Discord channel so that messages sent in one
166    /// place appear in the other; see [`linked_channel`](Lobby::linked_channel)
167    /// and [`LobbyMember::can_link_lobby`]. Because Discord permissions are not
168    /// synchronised into the game, *every* lobby member can read and write the
169    /// linked channel regardless of their Discord permissions — see
170    /// [`GuildChannel::is_viewable_and_writeable_by_all_members`].
171    ///
172    /// [`GuildChannel::is_viewable_and_writeable_by_all_members`]:
173    ///     crate::channel::GuildChannel::is_viewable_and_writeable_by_all_members
174    ///
175    /// # Staleness
176    ///
177    /// A handle references both the underlying data and the SDK instance.
178    /// Updates to the data are visible through existing handles without
179    /// re-creating them. If the SDK instance is destroyed while a handle is
180    /// still alive, every accessor returns a default value instead.
181    Lobby(sys::Discord_LobbyHandle) {
182        drop: sys::Discord_LobbyHandle_Drop,
183        clone: sys::Discord_LobbyHandle_Clone,
184    }
185}

187impl Lobby {
188    /// Information about the active voice call in this lobby, if there is one.
189    ///
190    /// Available even when the current user has not joined the call, so the
191    /// game can show which members are already in voice.
192    pub fn call_info(&self) -> Option<CallInfo> {
193        let mut raw = MaybeUninit::<sys::Discord_CallInfoHandle>::uninit();
194        // SAFETY: the out-parameter is only initialised when the call returns
195        // true, and ownership of the handle then transfers to us.
196        unsafe {
197            if sys::Discord_LobbyHandle_GetCallInfoHandle(self.raw_ptr(), raw.as_mut_ptr()) {
198                Some(CallInfo::from_raw(raw.assume_init()))
199            } else {
200                None
201            }
202        }
203    }
204
205    /// The member with the given user ID, if they belong to this lobby.
206    pub fn member(&self, member_id: u64) -> Option<LobbyMember> {
207        let mut raw = MaybeUninit::<sys::Discord_LobbyMemberHandle>::uninit();
208        // SAFETY: the out-parameter is only initialised when the call returns
209        // true, and ownership of the handle then transfers to us.
210        unsafe {
211            if sys::Discord_LobbyHandle_GetLobbyMemberHandle(
212                self.raw_ptr(),
213                member_id,
214                raw.as_mut_ptr(),
215            ) {
216                Some(LobbyMember::from_raw(raw.assume_init()))
217            } else {
218                None
219            }
220        }
221    }
222
223    /// The ID of the lobby.
224    pub fn id(&self) -> u64 {
225        // SAFETY: read-only call on a live handle.
226        unsafe { sys::Discord_LobbyHandle_Id(self.raw_ptr()) }
227    }
228
229    /// The Discord channel this lobby is linked to, if any.
230    pub fn linked_channel(&self) -> Option<LinkedChannel> {
231        let mut raw = MaybeUninit::<sys::Discord_LinkedChannel>::uninit();
232        // SAFETY: the out-parameter is only initialised when the call returns
233        // true, and ownership of the value then transfers to us.
234        unsafe {
235            if sys::Discord_LobbyHandle_LinkedChannel(self.raw_ptr(), raw.as_mut_ptr()) {
236                Some(LinkedChannel::from_raw(raw.assume_init()))
237            } else {
238                None
239            }
240        }
241    }
242
243    /// The user IDs of every member of this lobby.
244    ///
245    /// Cheaper than [`members`](Lobby::members) when only the IDs are needed.
246    pub fn member_ids(&self) -> Vec<u64> {
247        // SAFETY: the getter initialises the span and transfers ownership of the
248        // backing array; the elements are plain integers needing no wrapping.
249        unsafe {
250            span::out(
251                |out| sys::Discord_LobbyHandle_LobbyMemberIds(self.raw_ptr(), out),
252                |id| id,
253            )
254        }
255    }
256
257    /// A handle for every member of this lobby.
258    pub fn members(&self) -> Vec<LobbyMember> {
259        // SAFETY: the getter initialises the span and transfers ownership of
260        // both the array and every handle in it.
261        unsafe {
262            span::out(
263                |out| sys::Discord_LobbyHandle_LobbyMembers(self.raw_ptr(), out),
264                |raw| LobbyMember::from_raw(raw),
265            )
266        }
267    }
268
269    /// Developer-supplied metadata for this lobby.
270    ///
271    /// Simple string key/value pairs, a way to associate internal game state
272    /// with the lobby so that every member has easy access to it.
273    pub fn metadata(&self) -> HashMap<String, String> {
274        // SAFETY: the getter fully initialises the out-parameter and transfers
275        // ownership of the properties to us.
276        unsafe { properties_out(|out| sys::Discord_LobbyHandle_Metadata(self.raw_ptr(), out)) }
277    }
278}
