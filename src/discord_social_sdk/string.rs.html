1//! Conversions across the `Discord_String` boundary.
2//!
3//! The SDK uses two different ownership rules depending on direction:
4//!
5//! - **Into the SDK** (setters, arguments): the SDK copies immediately, so a
6//!   `Discord_String` borrowing Rust memory is sound as long as it only has to
7//!   survive the call. That is [`borrow`].
8//! - **Out of the SDK** (getters, callback payloads): the returned buffer is
9//!   freshly allocated and the caller owns it. It must be released with
10//!   `Discord_Free`. That is [`take`].
11//!
12//! `Discord_String` is *not* NUL-terminated, so it is always a `ptr`/`size`
13//! pair and never a `CStr`.
14
15use discord_social_sdk_sys as sys;
16use std::mem::MaybeUninit;
17
18/// Borrow a Rust string as a `Discord_String` for the duration of a call.
19///
20/// # Safety contract for callers
21///
22/// The result borrows `s`. It must not outlive `s`, and must only be passed to
23/// SDK functions that copy their input (which every setter does).
24pub(crate) fn borrow(s: &str) -> sys::Discord_String {
25    sys::Discord_String {
26        // Cast away const: the SDK treats this as read-only despite the mutable type.
27        ptr: s.as_ptr() as *mut u8,
28        size: s.len(),
29    }
30}
31
32/// Take ownership of a string returned by the SDK, freeing the SDK's buffer.
33///
34/// Invalid UTF-8 is replaced rather than rejected: the SDK sources these from
35/// user-authored content, and a lossy name is far better than a panic deep
36/// inside a callback.
37///
38/// # Safety
39///
40/// `s` must be a string the SDK just handed back and whose ownership was
41/// transferred to us. Calling this twice on the same value double-frees.
42pub(crate) unsafe fn take(s: sys::Discord_String) -> String {
43    if s.ptr.is_null() || s.size == 0 {
44        // A null pointer is never freeable; a zero-length buffer has nothing to read.
45        if !s.ptr.is_null() {
46            unsafe { sys::Discord_Free(s.ptr.cast()) };
47        }
48        return String::new();
49    }
50    let bytes = unsafe { std::slice::from_raw_parts(s.ptr, s.size) };
51    let out = String::from_utf8_lossy(bytes).into_owned();
52    unsafe { sys::Discord_Free(s.ptr.cast()) };
53    out
54}
55
56/// Read a borrowed `Discord_String` without taking ownership.
57///
58/// Used for strings delivered as callback arguments, which the SDK frees itself
59/// once the callback returns.
60///
61/// # Safety
62///
63/// `s` must point to a buffer valid for `s.size` bytes for the duration of the call.
64pub(crate) unsafe fn view(s: sys::Discord_String) -> String {
65    if s.ptr.is_null() || s.size == 0 {
66        return String::new();
67    }
68    let bytes = unsafe { std::slice::from_raw_parts(s.ptr, s.size) };
69    String::from_utf8_lossy(bytes).into_owned()
70}

72/// Call an SDK getter of the shape `void Get(self, Discord_String* out)`.
73///
74/// # Safety
75///
76/// `f` must fully initialise the out-parameter and transfer ownership of it.
77pub(crate) unsafe fn out<F>(f: F) -> String
78where
79    F: FnOnce(*mut sys::Discord_String),
80{
81    let mut raw = MaybeUninit::<sys::Discord_String>::uninit();
82    f(raw.as_mut_ptr());
83    unsafe { take(raw.assume_init()) }
84}
85
86/// Call an optional SDK getter of the shape `bool Get(self, Discord_String* out)`,
87/// where `false` means the field is absent and the out-parameter is untouched.
88///
89/// # Safety
90///
91/// Same contract as [`out`], but `f` only initialises the out-parameter when it
92/// returns `true`.
93pub(crate) unsafe fn out_opt<F>(f: F) -> Option<String>
94where
95    F: FnOnce(*mut sys::Discord_String) -> bool,
96{
97    let mut raw = MaybeUninit::<sys::Discord_String>::uninit();
98    if f(raw.as_mut_ptr()) {
99        Some(unsafe { take(raw.assume_init()) })
100    } else {
101        None
102    }
103}
104
105/// Run `f` with an optional borrowed string, matching the SDK's
106/// "null pointer means clear this field" convention on setters.
107pub(crate) fn with_opt<T, F>(value: Option<&str>, f: F) -> T
108where
109    F: FnOnce(*mut sys::Discord_String) -> T,
110{
111    match value {
112        Some(v) => {
113            let mut raw = borrow(v);
114            f(&mut raw)
115        }
116        None => f(std::ptr::null_mut()),
117    }
118}
